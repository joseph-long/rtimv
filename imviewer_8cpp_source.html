364 int imviewer::calcPixIndex(float pixval)
365 {
366  static float a = 1000;
367  static float log10_a = log10(a);
368 
369  //We first produce a value nominally between 0 and 1, though depending on the range it could be > 1.
370  pixval = (pixval - m_mindat)/((float)(m_maxdat-m_mindat));
371  if(pixval < 0) return 0;
372 
373  //Now we stretch it
374  switch(m_cbStretch)
375  {
376  case stretchLog:
377  pixval = log10(pixval*a+1)/log10_a;
378  break;
379  case stretchPow:
380  pixval = (pow(a, pixval))/a;
381  break;
382  case stretchSqrt:
383  pixval = sqrt(pixval);
384  break;
385  case stretchSquare:
386  pixval = pixval*pixval;
387  break;
388  default:
389  break;
390  }
391 
392  //Clamp it to <= 1
393  if(pixval > 1.) pixval = 1.;
394 
395  //And finally put it in the color bar index range
396  return pixval*(m_maxcol-m_mincol) + 0.5;
