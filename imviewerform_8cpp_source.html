470 
471  float mx = pt.x();

473 
474  if( mx < 0 || mx > qpmi->boundingRect().width() || my < 0 || my > qpmi->boundingRect().height() )
475  {

478

481  ui.graphicsView->textCoordX(mx-0.5);

484

489  idx_y = (int)(qpmi->boundingRect().height() - (my-0));
490  if(idx_y < 0) idx_y = 0;

492 
493  pixelF _pixel = pixel();
